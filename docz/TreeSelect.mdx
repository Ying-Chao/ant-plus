# TreeSelect 树选择

## 使用

```js
import { TreeSelect } from 'antx';
```

## 示例

<Playground>
  <TreeSelect
    data={[
      {
        id: 'tiger',
        name: '🐅 小脑斧',
        children: [{ id: 'little tiger', name: '🐅 小脑斧的儿子' }],
      },
      { id: 'dog', name: '🐶 小狗子', children: [{ id: 'little dog', name: '🐶 小狗子的儿子' }] },
      { id: 'cat', name: '🐱 小猫崽' },
    ]}
    keys={['id', 'name']}
    onChange={(value, label) => {
      console.log('value: ', value);
      console.log('label: ', label);
    }}
    search
    check
    back={TreeSelect.SHOW_ALL}
    msg="选个宠物吧"
    searchMsg="找一找"
  />
</Playground>
