# TreeSelect 树选择

## 使用

```js
import { TreeSelect } from 'antx';
```

## 示例

<Playground>
  <TreeSelect
    data={[
      {
        value: 'tiger',
        title: '🐅 小脑斧',
        children: [{ value: 'little tiger', title: '🐅 小脑斧的儿子' }],
      },
      {
        value: 'dog',
        title: '🐶 小狗子',
        children: [{ value: 'little dog', title: '🐶 小狗子的儿子' }],
      },
      { value: 'cat', title: '🐱 小猫崽' },
    ]}
    onChange={(value, selectedOptions) => {
      console.log('value: ', value);
      console.log('selectedOptions: ', selectedOptions);
    }}
    search
    check
    msg="选个宠物吧"
    searchMsg="找一找"
  />
</Playground>
